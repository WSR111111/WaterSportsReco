# DB 생성 및 연결

## DBeaver에 DB 생성
## 연결할 때 내가 보낸 설정 이미지를 참고

# 테이블 생성
```sql
-- 지역 관리 테이블
CREATE TABLE region (
    region_id INT AUTO_INCREMENT PRIMARY KEY,
    ldong_regn_cd VARCHAR(10) NOT NULL,
    ldong_regn_nm VARCHAR(100) NOT NULL,   
    ldong_sigungu_cd VARCHAR(10) UNIQUE, 
    ldong_sigungu_nm VARCHAR(100)
);

-- 스포츠 코드 테이블
CREATE TABLE sports (
    sport_id INT AUTO_INCREMENT PRIMARY KEY,
    sport_name VARCHAR(100) NOT NULL COMMENT '스포츠명',
    category_code VARCHAR(20) UNIQUE
);

-- 지표 코드 테이블
CREATE TABLE indicator (
    indicator_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL COMMENT '지표명',
    source ENUM('OCEAN','GROUND') NOT NULL COMMENT '값 출처'
);

-- 수상스포츠 사업장
CREATE TABLE leisure_place (
    leisure_id INT AUTO_INCREMENT PRIMARY KEY,
    cat3 VARCHAR(10) COMMENT '소분류 카테고리 (cat3)',
    content_id VARCHAR(20) UNIQUE,
    place_name VARCHAR(200) NOT NULL,
    address VARCHAR(300),
    address2 VARCHAR(500) COMMENT '상세주소 (addr2)',
    phone_number VARCHAR(50),
    latitude DECIMAL(10,6),
    longitude DECIMAL(10,6),
    area_code VARCHAR(10) COMMENT '지역코드 (areacode)',
    sigungu_code VARCHAR(10) COMMENT '시군구코드 (sigungucode)',
    first_image TEXT COMMENT '대표이미지 (firstimage)',
    first_image2 TEXT COMMENT '대표이미지2 (firstimage2)',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (sigungu_code) REFERENCES region(ldong_sigungu_cd),
    FOREIGN KEY (cat3) REFERENCES sports(category_code)
);

-- 사업장 상세 정보
CREATE TABLE place_detail (
    detail_id INT AUTO_INCREMENT PRIMARY KEY,
    content_id VARCHAR(20) NOT NULL COMMENT '콘텐츠 ID (한국관광공사)',
    homepage TEXT COMMENT '홈페이지 URL',
    overview TEXT COMMENT '개요/상세설명',
    FOREIGN KEY (content_id) REFERENCES leisure_place(content_id)
);

-- 지상 관측소 정보 테이블
CREATE TABLE ground_stations (
    station_id VARCHAR(20) PRIMARY KEY,          -- STN_ID (관측소 ID)
    station_name VARCHAR(100) NOT NULL,          -- STN_KO (관측소명 한글)
    latitude DECIMAL(10,6) NOT NULL,             -- LAT (위도)
    longitude DECIMAL(10,6) NOT NULL,            -- LON (경도)
    fct_id VARCHAR(20)                           -- FCT_ID (예보구역 ID)
);

-- 지상 관측 정보
CREATE TABLE ground_info (
    ground_info_id INT AUTO_INCREMENT PRIMARY KEY,
    station_id VARCHAR(20) NOT NULL,             -- ground_stations의 station_id 참조
    observation_time DATETIME NOT NULL,
    wind_speed DECIMAL(5,2),                     -- 풍속
    temp DECIMAL(5,2),                           -- 기온
    humidity DECIMAL(5,2),                       -- 습도
    FOREIGN KEY (station_id) REFERENCES ground_stations(station_id)
);

-- 해양 관측 정보
CREATE TABLE ocean_info (
    ocean_info_id INT AUTO_INCREMENT PRIMARY KEY,
    station_id VARCHAR(20) NOT NULL,             -- STN_ID
    station_name VARCHAR(100),                   -- STN_KO (관측소명)
    latitude DECIMAL(10,6),                      -- LAT (위도)
    longitude DECIMAL(10,6),                     -- LON (경도)
    observation_time DATETIME NOT NULL,          -- 관측 시간
    wave_height DECIMAL(5,2),                    -- WH (파고)
    water_temp DECIMAL(5,2)                      -- TW (수온)
);

-- 스포츠 요구 조건 (조건 정의 테이블)
CREATE TABLE sport_requirements (
    sport_id INT NOT NULL,
    indicator_id INT NOT NULL,
    min_value DECIMAL(6,2),
    max_value DECIMAL(6,2),
    PRIMARY KEY (sport_id, indicator_id),
    FOREIGN KEY (sport_id) REFERENCES sports(sport_id),
    FOREIGN KEY (indicator_id) REFERENCES indicator(indicator_id)
);
```

# 데이터 동기화

1.  backend 실행

2. 브라우저에서 http://localhost:8000/docs
 접속

3. POST API 엔드포인트들을 차례대로 Try it out → Execute 실행
→ 실행할 때마다 DB에 데이터가 저장됨
⚠️ place-details 엔드포인트는 실행 시간이 매우매우 오래걸림 주의
